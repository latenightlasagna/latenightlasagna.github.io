"DRAG TO THROW ME AROUND." : "TAP TO POKE ME."; } try { sceneRagdoll = new THREE.Scene(); const ambientLight = new THREE.AmbientLight(0xffffff, 0.6); sceneRagdoll.add(ambientLight); const dirLight = new THREE.DirectionalLight(0xffffff, 0.8); floorBody.addShape(new CANNON.Plane()); floorBody.quaternion.setFromAxisAngle(new CANNON.Vec3(1, 0, 0), -Math.PI / 2); world.addBody(floorBody); const torsoShape = new CANNON.Box(new CANNON.Vec3(0.5, 0.8, 0.5)); ragdollGroup = new THREE.Group(); sceneRagdoll.add(ragdollGroup); const loader = new THREE.GLTFLoader(); loader.load('./assets/simon-ps1.glb', (gltf) => { const characterRoot = gltf.scene; characterRoot.position.set(0, -0.8, 0); characterRoot.traverse((child) => { if (child.isMesh) { child.castShadow = true; child.receiveShadow = true; if (child.material.map) { child.material.map.magFilter = THREE.NearestFilter; child.material.map.minFilter = THREE.NearestFilter; } } if (child.isBone) { mixBones[child.name] = child; } }); ragdollGroup.add(characterRoot); }); // --- Desktop Drag & Mobile Poke Event Listeners --- window.addEventListener('pointerdown', (e) => { const rect = aboutCanvas.getBoundingClientRect(); if (rect.top < window.innerHeight && rect.bottom > 0 && torsoBody) { mouse.x = (e.clientX / window.innerWidth) * 2 - 1; mouse.y = -((e.clientY - rect.top) / rect.height) * 2 + 1; raycaster.setFromCamera(mouse, cameraRagdoll); aboutCanvas.style.cursor = 'grabbing'; // Create an invisible mathematical plane facing camera at the object's depth dragPlane.setFromNormalAndCoplanarPoint( cameraRagdoll.getWorldDirection(new THREE.Vector3()), ragdollGroup.position ); } else { // MOBILE: Poke mechanics const randomX = (Math.random() - 0.5) * 25; const randomZ = (Math.random() - 0.5) * 25; torsoBody.applyImpulse(new CANNON.Vec3(randomX, 45, randomZ), torsoBody.position); } } } }); window.addEventListener('pointermove', (e) => { if (!isDragging || !isDesktop) return; const rect = aboutCanvas.getBoundingClientRect(); mouse.x = (e.clientX / window.innerWidth) * 2 - 1; mouse.y = -((e.clientY - rect.top) / rect.height) * 2 + 1; // Find where the mouse intersects our invisible drag plane raycaster.setFromCamera(mouse, cameraRagdoll); raycaster.ray.intersectPlane(dragPlane, dragIntersection); }); window.addEventListener('pointerup', () => { if (isDragging) { isDragging = false; aboutCanvas.style.cursor = 'pointer'; } }); } catch (error) { console.error("Ragdoll setup failed", error); } // ========================================== // 3. GLOBAL RESIZE & ANIMATION LOOP // ========================================== window.addEventListener('resize', () => { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); trackballDummy.aspect = window.innerWidth / window.innerHeight; trackballDummy.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); if (cameraRagdoll && rendererRagdoll) { cameraRagdoll.aspect = window.innerWidth / window.innerHeight; cameraRagdoll.updateProjectionMatrix(); rendererRagdoll.setSize(window.innerWidth, window.innerHeight); } }); const scrollTrack = document.querySelector('.hero-scroll-track'); // CHANGED SCROLL TARGET TO ABOUT SECTION const targetSection = document.getElementById('about-section'); const scrollProgress = Math.max(0, Math.min(window.scrollY, maxScroll)) / maxScroll; camera.position.copy(trackballDummy.position).add(forwardVector.multiplyScalar(scrollProgress * 150)); renderer.render(scene, camera); } catch (error) {} // --- RAGDOLL SCENE (DRAG & FLOP PHYSICS) --- try { if (world && rendererRagdoll) { const aboutRect = aboutCanvas.getBoundingClientRect(); if (aboutRect.top < window.innerHeight && aboutRect.bottom > 0) { world.step(timeStep); if (mixBones.mixamorigHead) mixBones.mixamorigHead.rotation.set(0,0,0); } } rendererRagdoll.render(sceneRagdoll, cameraRagdoll); } } } catch (error) {} } animate();
